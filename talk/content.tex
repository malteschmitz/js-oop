\subsection{Objekte}

\begin{frame}[fragile]{Objekte}
  \begin{itemize}
    \item \emph{keine} Instanz einer Klasse
    \item dynamische Sammlung von Eigenschaften (Hash, Map, \ldots)
  \end{itemize}
    
  \begin{lstlisting}[gobble=4]
    // Objekt erzeugen
    var o = {};
    
    // Eigenschaft setzen
    o.foo = 42;
    
    // Eigenschaft auslesen
    console.log(o.foo);
    
    // Eigenschaft löschen
    delete o.foo;
  \end{lstlisting}  
\end{frame}
